fun int add(a : int, b : int) {
    return a + b;
}
result = add(2, 3);
print(result);
fun int factorial(n) {
    if (n <= 1) {
        return 1;
    }
    return n * factorial(n - 1);
}

f = factorial(5); // wynik: 120
print(f);
fun fun get_func() {
    return inc;
}

fun int inc(x : int) {
    return x + 1;
}

print(get_func()(7));

fun int ident(x:int) {
    return x;
}

fun int add1_decorator(f:fun, x:int) {
    return f(x + 1);
}

decorated = ident &*& add1_decorator;

result = decorated(7);
print(result);

fun int add3(a:int, b:int, c:int) {
    return a + b + c;
}

add_5_7 = (5, 7) =>> add3;
result = add_5_7(3);

print(result);   // 15
